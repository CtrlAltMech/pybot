@debug.no_return
def raiseAppropriateException(status_code):
    if status_code / 100 == 4:
        if status_code == 404:
            raise APINotFound()
        elif status_code == 401:
            raise APIUnauthorized()
        elif status_code == 400:
            raise APIBadCall()
        else:
            raise APIFailure()
    elif status_code / 100 == 5:
        raise APIError()
